# 二叉树

## 二叉树的前序遍历

### 题目

给定一个二叉树，返回它的 *前序* 遍历。

 **示例:**

```c++
输入: [1,null,2,3]  
   1
    \
     2
    /
   3 

输出: [1,2,3]
```

**进阶:** 递归算法很简单，你可以通过迭代算法完成吗？

### 题解

```c++
// 递归解法
// 先遍历当前节点，然后在遍历左节点，接着遍历右节点
class Solution {
public:
    vector<int> res;
    vector<int> preorderTraversal(TreeNode* root) {
        if (root != NULL) {
            res.push_back(root->val);
            preorderTraversal(root->left);
            preorderTraversal(root->right);
        }
        return res;
    }
};

// 非递归解法
class Solution {
public:
    vector<int> res;
    vector<int> preorderTraversal(TreeNode* root) {
        if (root == NULL) {
            return vector<int>();
        }
        stack<TreeNode *> st;
        st.push(root);
        while (!st.empty()) {
            TreeNode *ptr = st.top();
            st.pop();
            res.push_back(ptr->val);
            // 由于栈是逆序，所以这里应该先添加右节点，再添加左节点
            if (ptr->right != NULL) {
                st.push(ptr->right);
            }
            if (ptr->left != NULL) {
                st.push(ptr->left);
            }
        }
        return res;
    }
};
```

### 总结

- 递归和非递归两种方式

## 中序遍历二叉树

### 题目

给定一个二叉树，返回它的*中序* 遍历。

**示例:**

```c++
输入: [1,null,2,3]
   1
    \
     2
    /
   3

输出: [1,3,2]
```

**进阶:** 递归算法很简单，你可以通过迭代算法完成吗？

### 题解

```c++
// 递归方式
// 对于树中的任意节点来说，先打印它的左子树，然后再打印它本身，最后打印它的右子树。
class Solution {
public:
    vector<int> res;
    vector<int> inorderTraversal(TreeNode* root) {
        if (root != NULL) {
            inorderTraversal(root->left);
            res.push_back(root->val);
            inorderTraversal(root->right);
        }
        return res;
    }
};

// 迭代
class Solution {
public:
    vector<int> res;
    vector<int> inorderTraversal(TreeNode* root) {
        if (root == NULL) {
            return vector<int>();
        }
        stack<TreeNode *> st;
        TreeNode *ptr = root;
        /* 	1. 先到达最左边
			2. 把栈顶元素加入res中，然后把ptr指向栈顶元素的右节点
				2.1 如果右节点为nil，把ptr指向现在栈顶元素 
				2.2 如果右节点不为nil，把右节点压入栈中，一直查找到该节点的最后一个左节点
       	*/
        while (ptr != NULL || !st.empty()) {
            // 这里一进去，我们需要到达最左边节点，所以从根节点按顺序把节点压入栈中
            while (ptr != NULL) {
                st.push(ptr);
                ptr = ptr->left;
            }
            ptr = st.top();
            st.pop();
            res.push_back(ptr->val);
            ptr = ptr->right;
        }
        return res;
    }
};
```

## 后序遍历二叉树

### 题目

给定一个二叉树，返回它的 *后序* 遍历。

**示例:**

```c++
输入: [1,null,2,3]  
   1
    \
     2
    /
   3 

输出: [3,2,1]
```

**进阶:** 递归算法很简单，你可以通过迭代算法完成吗？

### 题解

```c++
// 递归方式
// 对于树中的任意节点来说，先打印它的左子树，然后再打印它的右子树，最后打印这个节点本身。
class Solution {
public:
    vector<int> res;
    vector<int> postorderTraversal(TreeNode* root) {
        if (root != NULL) {
            postorderTraversal(root->left);
            postorderTraversal(root->right);
            res.push_back(root->val);
        }
        return res;
    }
};

// 迭代
class Solution {
public:
    vector<int> postorderTraversal(TreeNode* root) {
       if (root == NULL) {
           return vector<int>();
        } 
        vector<int> res;
        stack<TreeNode *> st;
        TreeNode *ptr = root, *tag = NULL;
        while (ptr != NULL || !st.empty()) {
            // 到达二叉树最左边，并把沿途所有节点压入栈中
            // 每次当ptr指向右节点，把ptr和ptr的左节点全压入栈中
            while (ptr != NULL) {
                st.push(ptr);
                ptr = ptr->left;
            }
            ptr = st.top();
            st.pop();
            // 如果当前ptr右节点为NULL，把该ptr节点添加到res中，
            // 或者当tag == ptr->right，把该ptr节点添加到res中。
            // 如果当前ptr右节点不为NULL，把该ptr节点再次压入栈中
            if (ptr->right == NULL || tag == ptr->right) {
                res.push_back(ptr->val);
                // 用tag来标记遍历过的节点，如果该tag==ptr->right说明该ptr已经遍历过一次
                // 第一次遍历该ptr，由于右节点不为nil，所以把它再次压入栈中，遍历ptr的右节点
                // 当第二次指向该ptr，说明ptr的右节点说明已经遍历完了，所以就把该ptr添加到res
                tag = ptr;
                ptr = NULL;
            } else {
                st.push(ptr);
                ptr = ptr->right;
            }
        }
        return res;
    }
};
```
